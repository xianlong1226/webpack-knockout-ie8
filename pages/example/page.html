ko.msgBox(options) // 同message一样可以直接传入options对象： const options = { title: '提示', // 标题 message: '提示内容', // 内容 action: 'alert', // 弹窗类型 alert, confirm 或 prompt type: 'info', // 样式主题 info, warning, error, success showClose: true, // 是否显示关闭按钮 closeOnClickOutside: true, // 点击dialog外部关闭弹窗 closeOnPressEscape: true, // esc关闭弹窗 showConfirmButton: ko.observable(true), // 显示确定按钮 showCancelButton: ko.observable(false), // 显示取消按钮 confirmButtonText: ko.observable('确定'), // 确定按钮的文本 cancelButtonText: ko.observable('取消'), // 取消按钮的文本 beforeClose: function(){ // 关闭回调 console.log('关闭') } } // 同message一样，msgBox提供的方法返回值为promise对象，所以可以直接调用then方法传入取消和确定时的回调函数 // 封装的简化方法： ko.msgBox.alert('This is a message','提示',{}) // 普通提示信息 ko.msgBox.confirm('This is a message','请确认') // 需要确认时的提示信息 ko.msgBox.prompt('This is a message') // 需输入某个值时的提示信息 下列方法默认action为confirm，可以通过传入options修改 ko.msgBox.info('This is a message',{ action: 'alert' }) // 普通消息的提示信息 ko.msgBox.warning('This is a message') // 警告消息的提示信息 ko.msgBox.success('This is a message') // 成功后的提示信息 ko.msgBox.error('This is a message') // 错误的提示信息 // 第一个参数为消息内容，第二个参数为消息标题，第三个为自定义options，第二和第三参数可省略
